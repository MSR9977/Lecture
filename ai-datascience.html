بناء شبكة عصبية بسيطة

import torch
import torch.nn as nn

class NeuralNetwork(nn.Module):
def __init__(self):
super().__init__()
self.linear1 = nn.Linear(784, 128)
self.linear2 = nn.Linear(128, 10)
self.relu = nn.ReLU()

def forward(self, x):
x = self.relu(self.linear1(x))
return self.linear2(x)
