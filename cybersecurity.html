أنظمة التشغيل

إدارة Linux

# إدارة المستخدمين والصلاحيات
sudo useradd -m -s /bin/bash username
sudo chmod 755 /path/to/directory
sudo chown user:group file.txt

# إدارة الخدمات
systemctl status nginx
systemctl start apache2
systemctl enable postgresql

# مراقبة النظام
htop
df -h
free -m

Bash Scripting للأمان

#!/bin/bash
# سكريبت لفحص المنافذ المفتوحة
echo "Scanning open ports..."
netstat -tuln | grep LISTEN

# فحص المستخدمين المتصلين
echo "Active users:"
who

# فحص آخر محاولات تسجيل الدخول
echo "Last login attempts:"
last | head -10

PowerShell للأمان

# فحص حالة جدار الحماية
Get-NetFirewallProfile

# عرض العمليات المشبوهة
Get-Process | Where-Object {$_.CPU -gt 50}

# فحص التحديثات الأمنية
Get-HotFix | Sort-Object InstalledOn -Descending

# مراجعة سجلات الأحداث الأمنية
Get-EventLog -LogName Security -Newest 100
